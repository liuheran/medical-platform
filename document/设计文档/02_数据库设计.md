## 1. 数据库设计规范

### 1.1 命名规范

| 类型 | 规范 | 示例 |
|-----|------|-----|
| 表名 | 小写下划线，单数名词 | `sys_user`, `inquiry` |
| 字段名 | 小写下划线 | `user_id`, `created_at` |
| 索引名 | `idx_` + 字段名 | `idx_user_id` |
| 唯一索引 | `uk_` + 字段名 | `uk_phone` |
| 外键名 | `fk_` + 表名_字段 | `fk_doctor_user` |
| 视图名 | `v_` + 表名 | `v_user_detail` |
| 存储过程 | `sp_` + 功能 | `sp_generate_order` |
| 函数名 | `fn_` + 功能 | `fn_calculate_amount` |

### 1.2 字段设计规范

```sql
-- 时间字段统一命名
created_at    DATETIME  NOT NULL DEFAULT CURRENT_TIMESTAMP  COMMENT '创建时间'
updated_at    DATETIME  NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间'
deleted_at    DATETIME  COMMENT '删除时间(软删除)'

-- 主键统一使用 BIGINT UNSIGNED
id  BIGINT UNSIGNED NOT NULL AUTO_INCREMENT

-- 状态字段统一使用 TINYINT
status  TINYINT NOT NULL DEFAULT 1 COMMENT '状态:0禁用 1正常'

-- 外键统一使用 BIGINT UNSIGNED
user_id  BIGINT UNSIGNED NOT NULL

-- 金额统一使用 DECIMAL(10,2)
amount  DECIMAL(10,2) NOT NULL DEFAULT 0.00

-- 字符串长度合理分配
VARCHAR(20)   -- 手机号、验证码
VARCHAR(50)   -- 姓名、标题
VARCHAR(100)  -- 简介、地址
VARCHAR(255)  -- URL、邮箱
TEXT          -- 长文本、详情
JSON          -- JSON数据
```

### 1.3 索引设计原则

```
1. 索引数量控制: 单表索引不超过5个
2. 选择性原则: 区分度高的字段优先建索引
3. 最左前缀原则: 复合索引遵循最左前缀
4. 覆盖索引: 尽量使用覆盖索引减少回表
5. 避免索引失效: 不在索引列上使用函数
6. 主键顺序: 自增主键避免页分裂
```

## 3. ID 生成策略

### 3.1 分布式 ID 生成方案

```sql
-- 使用雪花算法生成唯一ID
-- Java实现: Twitter Snowflake
-- 64位 = 1位符号 + 41位时间戳 + 10位机器ID + 12位序列号

-- MySQL实现: 分布式表主键
CREATE TABLE `sequence` (
    `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    `biz_type` VARCHAR(50) NOT NULL COMMENT '业务类型',
    `current_value` BIGINT NOT NULL DEFAULT 0,
    `step` INT NOT NULL DEFAULT 1000,
    `version` INT NOT NULL DEFAULT 0,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_biz_type` (`biz_type`)
) ENGINE=InnoDB COMMENT='分布式ID序列表';

-- 获取ID的存储过程
DELIMITER //
CREATE PROCEDURE sp_next_id(
    IN p_biz_type VARCHAR(50),
    OUT p_result BIGINT
)
BEGIN
    DECLARE v_current_value BIGINT;
    DECLARE v_step INT;
    DECLARE v_version INT;

    START TRANSACTION;

    SELECT current_value, step, version
    INTO v_current_value, v_step, v_version
    FROM sequence
    WHERE biz_type = p_biz_type
    FOR UPDATE;

    UPDATE sequence
    SET current_value = current_value + step,
        version = version + 1
    WHERE biz_type = p_biz_type AND version = v_version;

    IF ROW_COUNT() = 0 THEN
        ROLLBACK;
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Concurrency conflict';
    ELSE
        COMMIT;
        SET p_result = v_current_value + v_step;
    END IF;
END //
DELIMITER ;
```

```java
// Java 加密工具类示例
@Service
public class EncryptionService {
    
    private static final String AES_KEY = "32位密钥..."; // 从配置中心获取
    
    /**
     * AES-256 加密
     */
    public String encrypt(String plaintext) {
        try {
            SecretKeySpec key = new SecretKeySpec(AES_KEY.getBytes(), "AES");
            Cipher cipher = Cipher.getInstance("AES/GCM/NoPadding");
            cipher.init(Cipher.ENCRYPT_MODE, key);
            byte[] encrypted = cipher.doFinal(plaintext.getBytes());
            return Base64.getEncoder().encodeToString(encrypted);
        } catch (Exception e) {
            throw new RuntimeException("加密失败", e);
        }
    }
    
    /**
     * SHA-256 哈希 (用于索引)
     */
    public String hash(String input) {
        return DigestUtils.sha256Hex(input);
    }
}
```

## 8. 缓存一致性设计方案

### 7.1 缓存问题与解决方案

| 问题 | 解决方案 | 适用场景 |
|-----|---------|---------|
| **缓存穿透** | 布隆过滤器 + 空值缓存 | 查询不存在的数据 |
| **缓存击穿** | 互斥锁 + 逻辑过期 | 热key过期瞬间大量请求 |
| **缓存雪崩** | 随机TTL + 过期时间打散 | 大量key同时过期 |
| **数据一致性** | Cache Aside + 延迟双删 | 读多写少场景 |

### 7.2 缓存架构设计

```yaml
# 缓存层级设计
L1: Caffeine (本地缓存)  # 热点数据，TTL 30秒
L2: Redis Cluster        # 分布式缓存，TTL 5分钟
L3: MySQL                # 最终数据源
```

### 7.3 缓存一致性实现

```java
@Service
public class DoctorCacheService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * Cache Aside 模式: 读流程
     */
    public Doctor getDoctor(Long doctorId) {
        String key = "doctor:" + doctorId;
        
        // 1. 查本地缓存
        Doctor doctor = localCache.get(key);
        if (doctor != null) return doctor;
        
        // 2. 查Redis
        doctor = (Doctor) redisTemplate.opsForValue().get(key);
        if (doctor != null) {
            localCache.put(key, doctor);
            return doctor;
        }
        
        // 3. 查数据库
        doctor = doctorMapper.selectById(doctorId);
        if (doctor != null) {
            redisTemplate.opsForValue().set(key, doctor, 5, TimeUnit.MINUTES);
            localCache.put(key, doctor);
        }
        return doctor;
    }
    
    /**
     * 延迟双删: 写流程
     */
    public void updateDoctor(Doctor doctor) {
        // 1. 删除缓存
        redisTemplate.delete("doctor:" + doctor.getId());
        
        // 2. 更新数据库
        doctorMapper.updateById(doctor);
        
        // 3. 延迟500ms再删除缓存（处理并发问题）
        try {
            Thread.sleep(500);
        } catch (InterruptedException e) {}
        redisTemplate.delete("doctor:" + doctor.getId());
    }
    
    /**
     * 布隆过滤器: 防止缓存穿透
     */
    public boolean mightExist(Long doctorId) {
        return bloomFilter.mightContain("doctor:" + doctorId);
    }
}
```

### 9.2 应用层校验实现

```java
@Service
public class InquiryService {
    
    @Autowired
    private UserFeignClient userClient;
    
    @Autowired
    private DoctorFeignClient doctorClient;
    
    /**
     * 创建问诊前校验
     */
    private void validateInquiry(Long userId, Long doctorId) {
        // 校验用户存在且状态正常
        UserDTO user = userClient.getUser(userId);
        if (user == null || user.getStatus() != 1) {
            throw new BusinessException("用户不存在或已被禁用");
        }
        
        // 校验医生存在且状态正常
        DoctorDTO doctor = doctorClient.getDoctor(doctorId);
        if (doctor == null || doctor.getStatus() != 1) {
            throw new BusinessException("医生不存在或已被禁用");
        }
        
        // 校验医生是否在线接诊
        if (doctor.getIsOnline() != 1 || doctor.getIsAccepting() != 1) {
            throw new BusinessException("医生当前不在线或停止接诊");
        }
    }
}
```

```sql
-- 每日校验逻辑外键一致性
SELECT 'doctor表存在user_id在sys_user中不存在的记录' AS check_name, COUNT(*) AS cnt
FROM doctor d
LEFT JOIN sys_user u ON d.user_id = u.id

LEFT JOIN doctor d ON i.doctor_id = d.id
```
